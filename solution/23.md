## 4. Code Implementation:

```
   # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        if not lists:
            return None

        def merge(l1, l2):  # LC21
            dummy = ListNode(0)
            cur = dummy
            while l1 and l2:
                if l1.val < l2.val:
                    cur.next = l1
                    l1 = l1.next
                else:
                    cur.next = l2
                    l2 = l2.next
                cur = cur.next
            cur.next = l1 or l2
            return dummy.next

        def divide(lists, left, right):
            if left == right:
                return lists[left]
            mid = (left + right) // 2
            list1 = divide(lists, left, mid)
            list2 = divide(lists, mid + 1, right)
            return merge(list1, list2)


```
